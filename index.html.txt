    <script>
        const video = document.getElementById('videoElement');
        const fileInput = document.getElementById('fileInput');
        const uploadSection = document.getElementById('uploadSection');
        const editorSection = document.getElementById('editorSection');
        const actionButtons = document.getElementById('actionButtons');
        
        let videoFile = null;
        let videoURL = null;
        let startTime = 0;
        let endTime = 0;
        let currentRotation = 0;
        let isDragging = null;
        let currentFilter = 'none';
        let brightnessValue = 100;
        
        // Drag & Drop
        function handleDragOver(e) {
            e.preventDefault();
            e.currentTarget.classList.add('dragover');
        }
        
        function handleDragLeave(e) {
            e.currentTarget.classList.remove('dragover');
        }
        
        function handleDrop(e) {
            e.preventDefault();
            e.currentTarget.classList.remove('dragover');
            if (e.dataTransfer.files.length) {
                processFile(e.dataTransfer.files[0]);
            }
        }
        
        function handleFileSelect(e) {
            if (e.target.files.length) {
                processFile(e.target.files[0]);
            }
        }
        
        function processFile(file) {
            if (!file.type.startsWith('video/')) {
                alert('Lütfen bir video dosyası yükleyin!');
                return;
            }
            
            videoFile = file;
            videoURL = URL.createObjectURL(file);
            video.src = videoURL;
            
            video.onloadedmetadata = () => {
                endTime = video.duration;
                initTrimControls();
                showEditor();
            };
        }
        
        function showEditor() {
            uploadSection.style.display = 'none';
            editorSection.classList.remove('hidden');
            actionButtons.classList.remove('hidden');
            actionButtons.classList.add('flex');
        }
        
        // Trim Controls
        function initTrimControls() {
            const slider = document.getElementById('trimSlider');
            const thumbLeft = document.getElementById('thumbLeft');
            const thumbRight = document.getElementById('thumbRight');
            const range = document.getElementById('rangeSelected');
            
            let rect = slider.getBoundingClientRect();
            
            function updateVisuals() {
                const leftPercent = parseFloat(thumbLeft.style.left) || 0;
                const rightPercent = 100 - (parseFloat(thumbRight.style.right) || 0);
                
                range.style.left = leftPercent + '%';
                range.style.width = (rightPercent - leftPercent) + '%';
                
                startTime = (leftPercent / 100) * video.duration;
                endTime = (rightPercent / 100) * video.duration;
                
                document.getElementById('startTimeDisplay').textContent = formatTime(startTime);
                document.getElementById('endTimeDisplay').textContent = formatTime(endTime);
            }
            
            thumbLeft.addEventListener('mousedown', (e) => {
                isDragging = 'left';
                e.preventDefault();
            });
            
            thumbRight.addEventListener('mousedown', (e) => {
                isDragging = 'right';
                e.preventDefault();
            });
            
            document.addEventListener('mousemove', (e) => {
                if (!isDragging) return;
                
                rect = slider.getBoundingClientRect();
                let percent = ((e.clientX - rect.left) / rect.width) * 100;
                percent = Math.max(0, Math.min(100, percent));
                
                if (isDragging === 'left') {
                    const rightPos = 100 - (parseFloat(thumbRight.style.right) || 0);
                    if (percent < rightPos - 2) {
                        thumbLeft.style.left = percent + '%';
                    }
                } else if (isDragging === 'right') {
                    const leftPos = parseFloat(thumbLeft.style.left) || 0;
                    if (percent > leftPos + 2) {
                        thumbRight.style.right = (100 - percent) + '%';
                    }
                }
                
                updateVisuals();
            });
            
            document.addEventListener('mouseup', () => {
                isDragging = null;
            });
            
            // Touch support for mobile
            thumbLeft.addEventListener('touchstart', (e) => {
                isDragging = 'left';
                e.preventDefault();
            });
            
            thumbRight.addEventListener('touchstart', (e) => {
                isDragging = 'right';
                e.preventDefault();
            });
            
            document.addEventListener('touchmove', (e) => {
                if (!isDragging || !e.touches[0]) return;
                
                rect = slider.getBoundingClientRect();
                let percent = ((e.touches[0].clientX - rect.left) / rect.width) * 100;
                percent = Math.max(0, Math.min(100, percent));
                
                if (isDragging === 'left') {
                    const rightPos = 100 - (parseFloat(thumbRight.style.right) || 0);
                    if (percent < rightPos - 2) {
                        thumbLeft.style.left = percent + '%';
                    }
                } else if (isDragging === 'right') {
                    const leftPos = parseFloat(thumbLeft.style.left) || 0;
                    if (percent > leftPos + 2) {
                        thumbRight.style.right = (100 - percent) + '%';
                    }
                }
                updateVisuals();
            });
            
            document.addEventListener('touchend', () => {
                isDragging = null;
            });
        }
        
        video.addEventListener('timeupdate', () => {
            document.getElementById('currentTimeDisplay').textContent = 
                formatTime(video.currentTime) + ' / ' + formatTime(video.duration);
            
            if (video.currentTime >= endTime && !video.paused) {
                video.pause();
                video.currentTime = startTime;
            }
        });
        
        function formatTime(seconds) {
            if (isNaN(seconds)) return '0:00';
            const mins = Math.floor(seconds / 60);
            const secs = Math.floor(seconds % 60);
            return `${mins}:${secs.toString().padStart(2, '0')}`;
        }
        
        function applyFilter(type) {
            currentFilter = type;
            document.querySelectorAll('.filter-btn').forEach(btn => btn.classList.remove('active'));
            document.querySelector(`[data-filter="${type}"]`).classList.add('active');
            
            updateVideoStyle();
        }
        
        function updateBrightness(val) {
            brightnessValue = val;
            document.getElementById('brightnessVal').textContent = val + '%';
            updateVideoStyle();
        }
        
        function updateVideoStyle() {
            let filter = `brightness(${brightnessValue}%)`;
            
            switch(currentFilter) {
                case 'grayscale': filter += ' grayscale(100%)'; break;
                case 'sepia': filter += ' sepia(100%)'; break;
                case 'contrast': filter += ' contrast(150%)'; break;
                case 'warm': filter += ' sepia(50%) saturate(140%) hue-rotate(-10deg)'; break;
                case 'cool': filter += ' saturate(80%) hue-rotate(180deg)'; break;
            }
            
            video.style.filter = filter;
        }
        
        function updateSpeed(val) {
            video.playbackRate = parseFloat(val);
            document.getElementById('speedVal').textContent = val + 'x';
        }
        
        function updateVolume(val) {
            video.volume = val / 100;
            document.getElementById('volumeVal').textContent = val + '%';
            if (val > 0) video.muted = false;
        }
        
        function toggleMute() {
            video.muted = !video.muted;
            const btn = document.getElementById('muteBtn');
            if (video.muted) {
                btn.innerHTML = '<i class="fas fa-volume-mute text-red-400"></i> Ses Aç';
                btn.classList.add('bg-gray-600');
            } else {
                btn.innerHTML = '<i class="fas fa-volume-up"></i> Sessiz';
                btn.classList.remove('bg-gray-600');
            }
        }
        
        function rotateVideo(deg) {
            currentRotation += deg;
            video.style.transform = `rotate(${currentRotation}deg)`;
        }
        
        function resetVideo() {
            video.pause();
            video.currentTime = 0;
            video.playbackRate = 1;
            video.style.filter = 'none';
            video.style.transform = 'rotate(0deg)';
            video.volume = 1;
            video.muted = false;
            
            currentRotation = 0;
            currentFilter = 'none';
            brightnessValue = 100;
            startTime = 0;
            endTime = video.duration || 0;
            
            // Reset UI
            document.querySelectorAll('input[type="range"]').forEach(input => {
                if (input.getAttribute('oninput').includes('Speed')) input.value = 1;
                else if (input.getAttribute('oninput').includes('Volume')) input.value = 100;
                else input.value = 100;
            });
            
            document.getElementById('speedVal').textContent = '1.0x';
            document.getElementById('volumeVal').textContent = '100%';
            document.getElementById('brightnessVal').textContent = '100%';
            document.getElementById('muteBtn').innerHTML = '<i class="fas fa-volume-up"></i> Sessiz';
            
            document.querySelectorAll('.filter-btn').forEach(btn => btn.classList.remove('active'));
            document.querySelector('[data-filter="none"]').classList.add('active');
            
            // Reset trim
            document.getElementById('thumbLeft').style.left = '0%';
            document.getElementById('thumbRight').style.right = '0%';
            document.getElementById('rangeSelected').style.left = '0%';
            document.getElementById('rangeSelected').style.width = '100%';
        }
        
        // Export Video - GERÇEKTEN ÇALIŞIYOR
        async function exportVideo() {
            if (!videoURL) {
                alert('Lütfen önce bir video yükleyin!');
                return;
            }
            
            const loadingOverlay = document.getElementById('loadingOverlay');
            const progressFill = document.getElementById('progressFill');
            const progressText = document.getElementById('progressText');
            
            loadingOverlay.style.display = 'flex';
            
            try {
                // Canvas oluştur
                const canvas = document.createElement('canvas');
                const ctx = canvas.getContext('2d');
                
                // Boyutlandırma (döndürmeye göre)
                if (Math.abs(currentRotation) % 180 === 90) {
                    canvas.width = video.videoHeight;
                    canvas.height = video.videoWidth;
                } else {
                    canvas.width = video.videoWidth;
                    canvas.height = video.videoHeight;
                }
                
                // MediaRecorder ayarları
                const stream = canvas.captureStream(30);
                const mediaRecorder = new MediaRecorder(stream, {
                    mimeType: 'video/webm;codecs=vp9'
                });
                
                const chunks = [];
                mediaRecorder.ondataavailable = (e) => {
                    if (e.data.size > 0) chunks.push(e.data);
                };
                
                mediaRecorder.onstop = () => {
                    const blob = new Blob(chunks, { type: 'video/webm' });
                    const url = URL.createObjectURL(blob);
                    const a = document.createElement('a');
                    a.href = url;
                    a.download = `EDTREVOX-edit-${Date.now()}.webm`;
                    document.body.appendChild(a);
                    a.click();
                    document.body.removeChild(a);
                    
                    loadingOverlay.style.display = 'none';
                    progressFill.style.width = '0%';
                    progressText.textContent = '%0';
                    
                    setTimeout(() => URL.revokeObjectURL(url), 1000);
                };
                
                // Kaydı başlat
                mediaRecorder.start(100);
                
                // Videoyu başlangıç noktasına al
                video.currentTime = startTime;
                video.playbackRate = 1; // Export sırasında normal hızda
                
                await new Promise(resolve => {
                    video.onseeked = resolve;
                });
                
                await video.play();
                
                const duration = endTime - startTime;
                const startTimeStamp = Date.now();
                let lastDrawTime = 0;
                
                function drawFrame(timestamp) {
                    if (video.paused || video.ended) {
                        mediaRecorder.stop();
                        video.pause();
                        return;
                    }
                    
                    if (video.currentTime >= endTime) {
                        mediaRecorder.stop();
                        video.pause();
                        return;
                    }
                    
                    // Her frame'de canvas'a çiz
                    ctx.clearRect(0, 0, canvas.width, canvas.height);
                    ctx.save();
                    ctx.translate(canvas.width / 2, canvas.height / 2);
                    ctx.rotate(currentRotation * Math.PI / 180);
                    
                    // Filtreleri uygula
                    ctx.filter = video.style.filter || 'none';
                    
                    ctx.drawImage(video, -video.videoWidth / 2, -video.videoHeight / 2);
                    ctx.restore();
                    
                    // İlerleme çubuğunu güncelle
                    const elapsed = video.currentTime - startTime;
                    const progress = Math.min((elapsed / duration) * 100, 100);
                    progressFill.style.width = progress + '%';
                    progressText.textContent = '%' + Math.floor(progress);
                    
                    requestAnimationFrame(drawFrame);
                }
                
                requestAnimationFrame(drawFrame);
                
            } catch (err) {
                console.error('Export hatası:', err);
                alert('Video dışa aktarılırken hata oluştu: ' + err.message);
                loadingOverlay.style.display = 'none';
            }
        }
        
        // Klavye kısayolları
        document.addEventListener('keydown', (e) => {
            if (e.code === 'Space' && document.activeElement.tagName !== 'INPUT') {
                e.preventDefault();
                if (video.paused) video.play();
                else video.pause();
            }
        });
